web_pages_base AS (
    SELECT
        p.TIMESTAMP, p.USER_ID, p.ANONYMOUS_ID, p.PARENT_TAXONOMY, p.NAME,
        p.SITE_SECTION, p.PREMIUM_CONTENT, p.PAYWALL_BARRIER, p.URL, p.PATH,
        NULLIF(TRIM(p.USER_ID), '')                                     AS user_id_norm,
        DATE_TRUNC('day',   p.TIMESTAMP::TIMESTAMP_NTZ)                 AS day_key,
        DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ)                 AS month_key,
        MAX(IFF(
            NULLIF(TRIM(p.USER_ID),'') IS NOT NULL
            AND LENGTH(NULLIF(TRIM(p.USER_ID),'')) <= 10
            AND REGEXP_LIKE(NULLIF(TRIM(p.USER_ID),''),'^[0-9]+$'),
            NULLIF(TRIM(p.USER_ID),''), NULL
        )) OVER (
            PARTITION BY DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ), p.ANONYMOUS_ID
        )                                                               AS stitched_user_id_month,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(p.TIMESTAMP::TIMESTAMP_NTZ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ),
                p.TIMESTAMP::TIMESTAMP_NTZ
            ) > 30
            OR LAG(p.TIMESTAMP::TIMESTAMP_NTZ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ) IS NULL
        ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.PAGES p
    WHERE p.TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
web_pages_stitched AS (
    SELECT *, CONCAT(ANONYMOUS_ID, '-', session_number::VARCHAR) AS session_id
    FROM web_pages_base
),

-- ============================================================
-- iOS / ANDROID SESSION DETECTION
-- ============================================================
ios_session_base AS (
    SELECT ORIGINAL_TIMESTAMP,
        COALESCE(NULLIF(TRIM(USER_ID),''), CONTEXT_DEVICE_ID) AS device_key,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP),
                ORIGINAL_TIMESTAMP) > 30
            OR LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) IS NULL
        ) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_IOS_LIVE.TRACKS
    WHERE ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
ios_sessions AS (
    SELECT device_key, ORIGINAL_TIMESTAMP, CONCAT(device_key,'-',session_number::VARCHAR) AS session_id
    FROM ios_session_base
),
android_session_base AS (
    SELECT ORIGINAL_TIMESTAMP,
        COALESCE(NULLIF(TRIM(USER_ID),''), CONTEXT_DEVICE_ID) AS device_key,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP),
                ORIGINAL_TIMESTAMP) > 30
            OR LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) IS NULL
        ) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_ANDROID_LIVE.TRACKS
    WHERE ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
android_sessions AS (
    SELECT device_key, ORIGINAL_TIMESTAMP, CONCAT(device_key,'-',session_number::VARCHAR) AS session_id
    FROM android_session_base
),

ios_total AS (
    SELECT DATE_TRUNC('day',t.ORIGINAL_TIMESTAMP) AS day, DATE_TRUNC('month',t.ORIGINAL_TIMESTAMP) AS month,
        'app' AS platform, NULLIF(TRIM(t.USER_ID),'') AS user_id_raw,
        COALESCE(NULLIF(TRIM(t.USER_ID),''),t.CONTEXT_DEVICE_ID) AS person_key,
        CASE WHEN NULLIF(TRIM(t.USER_ID),'') IS NOT NULL THEN NULLIF(TRIM(t.USER_ID),'')
             ELSE CONCAT('ios:',t.CONTEXT_DEVICE_ID) END AS monthly_key,
        IFF(NULLIF(TRIM(t.USER_ID),'') IS NOT NULL AND LENGTH(NULLIF(TRIM(t.USER_ID),''))<=10
            AND REGEXP_LIKE(NULLIF(TRIM(t.USER_ID),''),'^[0-9]+$'),TRUE,FALSE) AS is_logged_in,
        s.session_id
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_IOS_LIVE.TRACKS t
    LEFT JOIN ios_sessions s ON s.device_key=COALESCE(NULLIF(TRIM(t.USER_ID),''),t.CONTEXT_DEVICE_ID) AND s.ORIGINAL_TIMESTAMP=t.ORIGINAL_TIMESTAMP
    WHERE t.ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
android_total AS (
    SELECT DATE_TRUNC('day',t.ORIGINAL_TIMESTAMP) AS day, DATE_TRUNC('month',t.ORIGINAL_TIMESTAMP) AS month,
        'app' AS platform, NULLIF(TRIM(t.USER_ID),'') AS user_id_raw,
        COALESCE(NULLIF(TRIM(t.USER_ID),''),t.CONTEXT_DEVICE_ID) AS person_key,
        CASE WHEN NULLIF(TRIM(t.USER_ID),'') IS NOT NULL THEN NULLIF(TRIM(t.USER_ID),'')
             ELSE CONCAT('android:',t.CONTEXT_DEVICE_ID) END AS monthly_key,
        IFF(NULLIF(TRIM(t.USER_ID),'') IS NOT NULL AND LENGTH(NULLIF(TRIM(t.USER_ID),''))<=10
            AND REGEXP_LIKE(NULLIF(TRIM(t.USER_ID),''),'^[0-9]+$'),TRUE,FALSE) AS is_logged_in,
        s.session_id
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_ANDROID_LIVE.TRACKS t
    LEFT JOIN android_sessions s ON s.device_key=COALESCE(NULLIF(TRIM(t.USER_ID),''),t.CONTEXT_DEVICE_ID) AND s.ORIGINAL_TIMESTAMP=t.ORIGINAL_TIMESTAMP
    WHERE t.ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)

        w.session_id
    FROM web_pages_stitched w

web_anon_to_user_month AS (
    SELECT DISTINCT DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ) AS month_key, p.ANONYMOUS_ID,
        MAX(IFF(NULLIF(TRIM(p.USER_ID),'') IS NOT NULL AND LENGTH(NULLIF(TRIM(p.USER_ID),''))<=10
            AND REGEXP_LIKE(NULLIF(TRIM(p.USER_ID),''),'^[0-9]+$'),NULLIF(TRIM(p.USER_ID),''),NULL))
            OVER (PARTITION BY DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ), p.ANONYMOUS_ID) AS stitched_user_id_month
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.PAGES p
    WHERE p.TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
web_barrier_from_articles_raw AS (
    SELECT
        w.TIMESTAMP::TIMESTAMP_NTZ                                  AS event_ts,
        w.ANONYMOUS_ID,
        w.day_key                                                   AS day,
        w.month_key                                                 AS month,
        'web'                                                       AS platform,
        COALESCE(w.stitched_user_id_month, w.user_id_norm)          AS user_id_raw,
        COALESCE(w.stitched_user_id_month, w.ANONYMOUS_ID)          AS person_key,
        CASE WHEN w.stitched_user_id_month IS NOT NULL THEN w.stitched_user_id_month
             WHEN w.user_id_norm IS NOT NULL           THEN w.user_id_norm
             ELSE CONCAT('web:', w.ANONYMOUS_ID) END                AS monthly_key,
        IFF(COALESCE(w.stitched_user_id_month, w.user_id_norm) IS NOT NULL, TRUE, FALSE) AS is_logged_in,
        CASE WHEN w.NAME IN (SELECT v FROM rp_recommends_name_allowlist) THEN 'rp_recommends'
             WHEN w.PARENT_TAXONOMY='Premium content'               THEN 'premium_features'
             WHEN dm.desk IS NOT NULL                               THEN dm.desk
             WHEN w.SITE_SECTION IN (SELECT v FROM web_site_section_allowlist) THEN LOWER(w.SITE_SECTION)
             ELSE 'news' END                                        AS desk
    FROM web_pages_stitched w
    LEFT JOIN web_desk_map dm ON dm.parent_taxonomy=w.PARENT_TAXONOMY
    WHERE w.TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    AND COALESCE(w.PAYWALL_BARRIER,FALSE)=TRUE
    AND (w.NAME IN (SELECT v FROM web_name_allowlist) OR w.PARENT_TAXONOMY IN (SELECT v FROM web_parent_tax_allowlist) OR w.SITE_SECTION IN (SELECT v FROM web_site_section_allowlist))
),
web_barrier_from_table_raw AS (
    SELECT
        b.TIMESTAMP::TIMESTAMP_NTZ                                  AS event_ts,
        b.ANONYMOUS_ID,
        DATE_TRUNC('day',b.TIMESTAMP::TIMESTAMP_NTZ)                AS day,
        DATE_TRUNC('month',b.TIMESTAMP::TIMESTAMP_NTZ)              AS month,
        'web'                                                       AS platform,
        COALESCE(m.stitched_user_id_month,NULLIF(TRIM(b.USER_ID),'')) AS user_id_raw,
        COALESCE(m.stitched_user_id_month, b.ANONYMOUS_ID)          AS person_key,
        CASE WHEN COALESCE(m.stitched_user_id_month,NULLIF(TRIM(b.USER_ID),'')) IS NOT NULL
             THEN COALESCE(m.stitched_user_id_month,NULLIF(TRIM(b.USER_ID),''))
             ELSE CONCAT('web:',b.ANONYMOUS_ID) END                 AS monthly_key,
        IFF(COALESCE(m.stitched_user_id_month,NULLIF(TRIM(b.USER_ID),'')) IS NOT NULL,TRUE,FALSE) AS is_logged_in,
        CASE WHEN b.SITE_SECTION='News'       THEN 'news'
             WHEN b.SITE_SECTION='Bloodstock' THEN 'bloodstock'
             WHEN b.SITE_SECTION='Tipping'    THEN 'tipping'
             ELSE NULL END                                          AS desk
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.BARRIER b
    LEFT JOIN web_anon_to_user_month m ON m.month_key=DATE_TRUNC('month',b.TIMESTAMP::TIMESTAMP_NTZ) AND m.ANONYMOUS_ID=b.ANONYMOUS_ID
    WHERE b.TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    AND b.SITE_SECTION IN ('News','Bloodstock','Tipping')
),
web_barrier_deduped AS (

        ANONYMOUS_ID,
        DATE_TRUNC('minute', event_ts)                              AS event_minute,
        day, month, platform, user_id_raw, person_key, monthly_key, is_logged_in, desk
    FROM (
        SELECT event_ts, ANONYMOUS_ID, day, month, platform,
               user_id_raw, person_key, monthly_key, is_logged_in, desk
        FROM web_barrier_from_articles_raw

    SELECT DISTINCT month::DATE AS month_start, monthly_key FROM total_base

    UNION SELECT DISTINCT month::DATE, monthly_key FROM barrier_base
    UNION SELECT DISTINCT month::DATE, monthly_key FROM all_clicks_raw
    UNION SELECT DISTINCT month::DATE, monthly_key FROM traffic_tc_and_recs_raw

persona_candidates AS (
    SELECT month_start, monthly_key, 'ios' AS src_platform, REPLACE(monthly_key,'ios:','') AS unified_id
    FROM distinct_monthly_keys WHERE LEFT(monthly_key,4) = 'ios:'
    UNION ALL SELECT month_start, monthly_key, 'android', REPLACE(monthly_key,'android:','')
    FROM distinct_monthly_keys WHERE LEFT(monthly_key,8) = 'android:'
    UNION ALL SELECT month_start, monthly_key, 'web', REPLACE(monthly_key,'web:','')
    FROM distinct_monthly_keys WHERE LEFT(monthly_key,4) = 'web:'
    UNION ALL SELECT month_start, monthly_key, 'ios', monthly_key
    FROM distinct_monthly_keys WHERE REGEXP_LIKE(monthly_key,'^[0-9]+$')
    UNION ALL SELECT month_start, monthly_key, 'android', monthly_key
    FROM distinct_monthly_keys WHERE REGEXP_LIKE(monthly_key,'^[0-9]+$')
    UNION ALL SELECT month_start, monthly_key, 'web', monthly_key
    FROM distinct_monthly_keys WHERE REGEXP_LIKE(monthly_key,'^[0-9]+$')

    SELECT x.month_start, x.monthly_key,
        IFF(COALESCE(x.persona,'Unclassified') IN ('Bots',''), 'Unclassified', COALESCE(x.persona,'Unclassified')) AS persona
    FROM (
        SELECT pc.month_start, pc.monthly_key, pm.persona, pm.persona_date,
            ROW_NUMBER() OVER (PARTITION BY pc.month_start, pc.monthly_key ORDER BY pm.persona_date DESC NULLS LAST) AS rn
        FROM persona_candidates pc
        LEFT JOIN persona_map pm ON pm.month_start=pc.month_start AND pm.src_platform=pc.src_platform AND pm.unified_id=pc.unified_id
    ) x WHERE x.rn = 1

        SELECT period_start,grain,platform,persona,desk,total_ftds_from_editorial,total_ftds_from_recs FROM ftd_by_desk_month
        UNION ALL SELECT period_start,grain,platform,persona,desk,total_ftds_from_editorial,total_ftds_from_recs FROM ftd_total_platform_month

        UNION ALL SELECT period_start,grain,platform,persona,desk,total_ftds_from_editorial,total_ftds_from_recs FROM ftd_all_editorial_total_day

    SELECT month_start AS period_start,'total', persona, desk,
        COUNT(*)::NUMBER(18,0)

    FROM subscriber_active_days_base GROUP BY 1,3

subscriber_ultimate_daily AS (

        persona,
        desk,
        source,
        COUNT(*)::NUMBER(18,0) AS clicks
    FROM (
        SELECT
            c.ORIGINAL_TIMESTAMP::TIMESTAMP_NTZ AS click_ts,
            'web' AS platform,
            'rp_recommends' AS desk,
            'TC_ARTICLE_CLICK' AS source,
            COALESCE(pml.persona,'Unclassified') AS persona
        FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.TC_ARTICLE_CLICK c
        LEFT JOIN persona_monthly_latest pml
          ON pml.month_start=DATE_TRUNC('month',c.ORIGINAL_TIMESTAMP)::DATE
         AND pml.monthly_key=COALESCE(NULLIF(TRIM(c.USER_ID::STRING),''), CONCAT('web:',c.ANONYMOUS_ID))

        UNION ALL
        SELECT
            c.ORIGINAL_TIMESTAMP::TIMESTAMP_NTZ,'web','rp_recommends','ALL_OFFERS_CLICK',
            COALESCE(pml.persona,'Unclassified')
        FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.ALL_OFFERS_CLICK c
        LEFT JOIN persona_monthly_latest pml
          ON pml.month_start=DATE_TRUNC('month',c.ORIGINAL_TIMESTAMP)::DATE
         AND pml.monthly_key=COALESCE(NULLIF(TRIM(c.USER_ID::STRING),''), CONCAT('web:',c.ANONYMOUS_ID))

        UNION ALL
        SELECT
            c.ORIGINAL_TIMESTAMP::TIMESTAMP_NTZ,'web','rp_recommends','FREE_BETS_TAP',
            COALESCE(pml.persona,'Unclassified')
        FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.FREE_BETS_TAP c
        LEFT JOIN persona_monthly_latest pml
          ON pml.month_start=DATE_TRUNC('month',c.ORIGINAL_TIMESTAMP)::DATE
         AND pml.monthly_key=COALESCE(NULLIF(TRIM(c.USER_ID::STRING),''), CONCAT('web:',c.ANONYMOUS_ID))
    ) x
    GROUP BY 1,2,3,4,5
),
clicks_source_json AS (
    SELECT
        period_start,'month' AS grain,platform,persona,desk,
        OBJECT_AGG(source, clicks) AS editorial_clicks_by_source
    FROM clicks_source_month
    GROUP BY 1,2,3,4,5
),
ftd_source_json AS (
    SELECT
        period_start,'month' AS grain,platform,persona,desk,
        OBJECT_AGG(source, ftds) AS ftds_by_source
    FROM (
        SELECT
            c.period_start,c.platform,c.persona,c.desk,c.source,
            COUNT(DISTINCT c.racingpost_id)::NUMBER(18,0) AS ftds
        FROM (
            SELECT
                DATE_TRUNC('month', click_ts)::DATE AS period_start,
                platform,persona,desk,source,racingpost_id,MIN(click_ts) AS first_click_ts
            FROM editorial_clicks
            WHERE racingpost_id IS NOT NULL
            GROUP BY 1,2,3,4,5,6
        ) c
        JOIN ftd_deduped f
          ON f.month_start=c.period_start
         AND f.racingpost_id=c.racingpost_id
         AND f.first_deposit_ts>c.first_click_ts
        GROUP BY 1,2,3,4,5
    ) z
    GROUP BY 1,2,3,4,5
)

SELECT
    m.grain, m.period_start, m.platform, m.persona, 
    CASE
  WHEN m.desk = 'ALL_EDITORIAL'    THEN 'All Editorial'
  WHEN m.desk = 'All Editorial'    THEN 'All Editorial'
  WHEN m.desk = 'all'              THEN 'All'
  WHEN m.desk = 'All'              THEN 'All'
  WHEN m.desk = 'news'             THEN 'News'
  WHEN m.desk = 'tipping'          THEN 'Tipping'
  WHEN m.desk = 'rp_recommends'    THEN 'RP Recommends'
  WHEN m.desk = 'premium_features' THEN 'Premium Features'
  WHEN m.desk = 'bloodstock'       THEN 'Bloodstock'
  WHEN m.desk = 'sport'            THEN 'Sport'
  ELSE INITCAP(REPLACE(m.desk,'_',' '))
END AS desk,
    m.users, m.users AS desk_audience_users,
    m.events, m.users_logged_in, m.events_logged_in,
    m.users_subscriber, m.events_subscriber, m.users_free, m.events_free,
    m.users_premium, m.events_premium, m.users_barrier, m.barrier_events,
    n.net_subscribers,
    IFF(m.grain='month', ad.total_active_days, NULL)                        AS total_active_days,
    IFF(m.grain='month' AND m.desk!='All',
        ROUND(CASE WHEN COALESCE(m.users_subscriber,0)=0 THEN NULL
              ELSE m.users_premium/m.users_subscriber*100 END,2),
        NULL)                                                                AS pct_subscribers_consuming_premium,
    IFF(m.desk!='All', ftd.total_ftds_from_editorial,      NULL)            AS total_ftds_from_editorial,
    IFF(m.desk!='All', ftd.total_ftds_from_tc,             NULL)            AS total_ftds_from_tc,
    IFF(m.desk!='All', ftd.total_ftds_from_recs,           NULL)            AS total_ftds_from_recs,
    IFF(m.desk!='All', ftd.total_ftds_from_tc_and_recs,    NULL)            AS total_ftds_from_tc_and_recs,
    IFF(m.desk!='All', ftd.total_ftds_from_editorial_plus, NULL)            AS total_ftds_from_editorial_plus,
    cl.total_clicks_from_editorial,
    tr.traffic_events_to_tc_and_recs,
    tv.traffic_visitors_to_tc_and_recs,
    m.users_subscriber_ultimate, m.users_subscriber_tipping,
    m.is_subscriber, m.editorial_subscribers, m.subscriber_flag,
    
    IFF(m.grain='month' AND m.platform IN ('web','app') AND m.desk IN ('ALL_EDITORIAL','All Editorial'),
    um.daily_avg_pct_ultimate_consuming_editorial, NULL)::NUMBER(18,2)
    AS daily_avg_pct_ultimate_consuming_editorial,
    IFF(m.grain='month' AND m.platform IN ('web','app') AND m.desk IN ('ALL_EDITORIAL','All Editorial'),
    tm.daily_avg_pct_tipping_consuming_editorial, NULL)::NUMBER(18,2)
    AS daily_avg_pct_tipping_consuming_editorial,

    IFF(m.grain='day',
        ROUND(AVG(m.pageviews) OVER (
            PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
            ORDER BY m.period_start ROWS BETWEEN 12 PRECEDING AND CURRENT ROW),2),
        m.pageviews)::NUMBER(18,0)                                           AS pageviews,
    IFF(m.grain='month' AND m.desk IN ('ALL_EDITORIAL','All Editorial')
        AND COALESCE(m.editorial_subscribers,0) > 0,
        ROUND(ad.total_active_days / NULLIF(m.editorial_subscribers,0), 2),
        NULL)::NUMBER(18,2)
        AS active_days_per_editorial_subscriber,
    m.unique_sessions,
    ROUND(m.pageviews::FLOAT / NULLIF(m.unique_sessions,0), 2)::NUMBER(18,2)
        AS pageviews_per_unique_session,
    IFF(m.grain='day',
        ROUND(AVG(m.pageviews::FLOAT / NULLIF(m.unique_sessions,0)) OVER (
            PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
            ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),2),
        NULL)::NUMBER(18,2)
        AS pageviews_per_session_dow_avg,
    IFF(m.grain='day',
        ROUND((m.pageviews::FLOAT / NULLIF(m.unique_sessions,0))
            - AVG(m.pageviews::FLOAT / NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),2),
        NULL)::NUMBER(18,2)
        AS pageviews_vs_dow_avg,
    IFF(m.grain='day',
        ROUND(((m.pageviews::FLOAT / NULLIF(m.unique_sessions,0))
            - AVG(m.pageviews::FLOAT / NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING))
            / NULLIF(AVG(m.pageviews::FLOAT / NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),0)*100,2),
        NULL)::NUMBER(18,2)
        AS pageviews_pct_vs_dow_avg,
    m.premium_editorial_users,
    m.free_editorial_users,
    TO_VARCHAR(cs.editorial_clicks_by_source) AS editorial_clicks_by_source,
TO_VARCHAR(fs.ftds_by_source)             AS ftds_by_source
FROM (
    SELECT * FROM total_metrics
    UNION ALL SELECT * FROM editorial_metrics_by_desk
    UNION ALL SELECT * FROM editorial_metrics_all_editorial
) m
LEFT JOIN net_subscriber_calc n
    ON  n.grain        = m.grain
    AND n.period_start = m.period_start
    AND n.platform     = m.platform
    AND n.persona      = m.persona
    AND n.desk         = m.desk
LEFT JOIN active_days_all ad
    ON  ad.period_start = m.period_start
    AND ad.platform     = m.platform
    AND ad.persona      = m.persona
    AND ad.desk         = m.desk
LEFT JOIN ftd_all ftd
    ON  ftd.grain        = m.grain
    AND ftd.period_start = m.period_start
    AND ftd.platform     = m.platform
    AND ftd.persona      = m.persona
    AND ftd.desk         = m.desk
LEFT JOIN clicks_all cl
    ON  cl.grain        = m.grain
    AND cl.period_start = m.period_start
    AND cl.platform     = m.platform
    AND cl.persona      = m.persona
    AND cl.desk         = m.desk
LEFT JOIN traffic_all tr
    ON  tr.grain        = m.grain
    AND tr.period_start = m.period_start
    AND tr.platform     = m.platform
    AND tr.persona      = m.persona
    AND tr.desk         = m.desk
LEFT JOIN ultimate_month_daily_avg um
  ON um.period_start=m.period_start AND um.platform=m.platform
LEFT JOIN tipping_month_daily_avg tm
  ON tm.period_start=m.period_start AND tm.platform=m.platform
LEFT JOIN traffic_visitors_all tv
  ON tv.grain = m.grain
 AND tv.period_start = m.period_start
 AND tv.platform = m.platform
 AND tv.persona = m.persona
 AND tv.desk = m.desk
LEFT JOIN clicks_source_json cs
  ON cs.grain=m.grain
 AND cs.period_start=m.period_start
 AND cs.platform=m.platform
 AND cs.persona=m.persona
 AND cs.desk=m.desk

LEFT JOIN ftd_source_json fs
  ON fs.grain=m.grain
 AND fs.period_start=m.period_start
 AND fs.platform=m.platform
 AND fs.persona=m.persona
 AND fs.desk=m.desk
ORDER BY grain, period_start, platform, persona, desk;
