web_pages_base AS (
    SELECT
        p.TIMESTAMP, p.USER_ID, p.ANONYMOUS_ID, p.PARENT_TAXONOMY, p.NAME,
        p.SITE_SECTION, p.PREMIUM_CONTENT, p.PAYWALL_BARRIER, p.URL, p.PATH,
        NULLIF(TRIM(p.USER_ID), '')                                     AS user_id_norm,
        DATE_TRUNC('day',   p.TIMESTAMP::TIMESTAMP_NTZ)                 AS day_key,
        DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ)                 AS month_key,
        MAX(IFF(
            NULLIF(TRIM(p.USER_ID),'') IS NOT NULL
            AND LENGTH(NULLIF(TRIM(p.USER_ID),'')) <= 10
            AND REGEXP_LIKE(NULLIF(TRIM(p.USER_ID),''),'^[0-9]+$'),
            NULLIF(TRIM(p.USER_ID),''), NULL
        )) OVER (
            PARTITION BY DATE_TRUNC('month', p.TIMESTAMP::TIMESTAMP_NTZ), p.ANONYMOUS_ID
        )                                                               AS stitched_user_id_month,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(p.TIMESTAMP::TIMESTAMP_NTZ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ),
                p.TIMESTAMP::TIMESTAMP_NTZ
            ) > 30
            OR LAG(p.TIMESTAMP::TIMESTAMP_NTZ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ) IS NULL
        ) OVER (PARTITION BY p.ANONYMOUS_ID ORDER BY p.TIMESTAMP::TIMESTAMP_NTZ) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_WEB_LIVE.PAGES p
    WHERE p.TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
web_pages_stitched AS (
    SELECT *, CONCAT(ANONYMOUS_ID, '-', session_number::VARCHAR) AS session_id
    FROM web_pages_base
),

-- ============================================================
-- iOS / ANDROID SESSION DETECTION
-- ============================================================
ios_session_base AS (
    SELECT ORIGINAL_TIMESTAMP,
        COALESCE(NULLIF(TRIM(USER_ID),''), CONTEXT_DEVICE_ID) AS device_key,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP),
                ORIGINAL_TIMESTAMP) > 30
            OR LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) IS NULL
        ) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_IOS_LIVE.TRACKS
    WHERE ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
ios_sessions AS (
    SELECT device_key, ORIGINAL_TIMESTAMP, CONCAT(device_key,'-',session_number::VARCHAR) AS session_id
    FROM ios_session_base
),
android_session_base AS (
    SELECT ORIGINAL_TIMESTAMP,
        COALESCE(NULLIF(TRIM(USER_ID),''), CONTEXT_DEVICE_ID) AS device_key,
        CONDITIONAL_TRUE_EVENT(
            DATEDIFF('minute',
                LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP),
                ORIGINAL_TIMESTAMP) > 30
            OR LAG(ORIGINAL_TIMESTAMP) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) IS NULL
        ) OVER (PARTITION BY COALESCE(NULLIF(TRIM(USER_ID),''),CONTEXT_DEVICE_ID) ORDER BY ORIGINAL_TIMESTAMP) AS session_number
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_ANDROID_LIVE.TRACKS
    WHERE ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
),
android_sessions AS (
    SELECT device_key, ORIGINAL_TIMESTAMP, CONCAT(device_key,'-',session_number::VARCHAR) AS session_id
    FROM android_session_base
),

        IFF(COALESCE(w.stitched_user_id_month, w.user_id_norm) IS NOT NULL, TRUE, FALSE) AS is_logged_in,

    SELECT month_start, monthly_key, 'ios' AS src_platform, REPLACE(monthly_key,'ios:','') AS unified_id
    FROM distinct_monthly_keys WHERE LEFT(monthly_key,4)='ios:'
    UNION ALL SELECT month_start,monthly_key,'android',REPLACE(monthly_key,'android:','') FROM distinct_monthly_keys WHERE LEFT(monthly_key,8)='android:'
    UNION ALL SELECT month_start,monthly_key,'web',REPLACE(monthly_key,'web:','') FROM distinct_monthly_keys WHERE LEFT(monthly_key,4)='web:'
    UNION ALL SELECT month_start,monthly_key,'ios',monthly_key FROM distinct_monthly_keys WHERE REGEXP_LIKE(monthly_key,'^[0-9]+$')
    UNION ALL SELECT month_start,monthly_key,'android',monthly_key FROM distinct_monthly_keys WHERE REGEXP_LIKE(monthly_key,'^[0-9]+$')

    FROM (
        SELECT pc.month_start,pc.monthly_key,pm.persona,pm.persona_date,
            ROW_NUMBER() OVER (PARTITION BY pc.month_start,pc.monthly_key ORDER BY pm.persona_date DESC NULLS LAST) AS rn
        FROM persona_candidates pc
        LEFT JOIN persona_map pm ON pm.month_start=pc.month_start AND pm.src_platform=pc.src_platform AND pm.unified_id=pc.unified_id
    ) x WHERE x.rn=1

    SELECT
        a.DATE::DATE                                                    AS period_start,
        DATE_TRUNC('month', a.DATE)::DATE                               AS month_start,
        CASE
            WHEN UPPER(TRIM(b."Category 1")) IN ('WEB','MOBILE WEB')   THEN 'web'
            WHEN UPPER(TRIM(b."Category 1")) = 'APP'                   THEN 'app'
            ELSE 'total'
        END                                                             AS platform,
        CASE
            WHEN TRIM(b."Category 2") = 'Transactional Content'        THEN 'rp_recommends'
            WHEN TRIM(b."Category 3") IN (
                'RP Recommends','RP Recommends EG',
                'RP Reccommends','RP Reccommends EG'
            )                                                           THEN 'rp_recommends'
            WHEN TRIM(b."Category 2") = 'Content'                      THEN 'news'
            ELSE 'other'
        END                                                             AS desk,
        TRIM(b."Category 2")                                            AS channel,
        TRIM(b."Category 3")                                            AS source,
        SUM(a.NEW_DEPOSITORS)                                           AS ftd
    FROM SSG_EXTRA.TBL_FTD_ACCOUNTS_MASTER a
    JOIN REPORTING.SSG_EXTRA.TBL_STAGING_FTD_ACCOUNTS_ALL_LINKS b
      ON a.LINK_ID=b.LINKS AND a.BOOKMAKER=b."Bookmaker"
    WHERE a.DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    GROUP BY 1,2,3,4,5,6

-- AVERAGE ENGAGED TIME PER STORY
-- (total time spent on story / total distinct visitors per story, then averaged)
-- App only: iOS + Android (web has no screen-in/out events)
-- ============================================================
ios_article_time_raw AS (
    SELECT
        DATE_TRUNC('day',  s.ORIGINAL_TIMESTAMP)::DATE          AS day,
        DATE_TRUNC('month', s.ORIGINAL_TIMESTAMP)::DATE         AS month,
        'app'                                                    AS platform,
        s.ARTICLE_ID::VARCHAR                                    AS story_id,
        COALESCE(NULLIF(TRIM(s.USER_ID),''), s.CONTEXT_DEVICE_ID) AS visitor_key,
        CASE WHEN NULLIF(TRIM(s.USER_ID),'') IS NOT NULL
             THEN NULLIF(TRIM(s.USER_ID),'')
             ELSE CONCAT('ios:', s.CONTEXT_DEVICE_ID) END        AS monthly_key,
        s.ORIGINAL_TIMESTAMP,
        s.PARENT_TAXONOMY,                                       -- ← ADD THIS
        DATEDIFF('second',
            s.ORIGINAL_TIMESTAMP,
            MIN(t.ORIGINAL_TIMESTAMP)
        )                                                        AS time_on_screen_seconds
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_IOS_LIVE.SCREEN_ARTICLE s
    JOIN SEGMENT_EVENTS.RACING_POST_JANUS_IOS_LIVE.TRACKS t
        ON  COALESCE(NULLIF(TRIM(t.USER_ID),''), t.CONTEXT_DEVICE_ID)
          = COALESCE(NULLIF(TRIM(s.USER_ID),''), s.CONTEXT_DEVICE_ID)
        AND t.ORIGINAL_TIMESTAMP > s.ORIGINAL_TIMESTAMP
        AND DATEDIFF('minute', s.ORIGINAL_TIMESTAMP, t.ORIGINAL_TIMESTAMP) <= 60
    WHERE s.ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    AND s.ARTICLE_ID IS NOT NULL
    AND s.PARENT_TAXONOMY IN (SELECT v FROM app_parent_tax_allowlist)
    GROUP BY 1,2,3,4,5,6,s.ORIGINAL_TIMESTAMP,s.PARENT_TAXONOMY  -- ← ADD s.PARENT_TAXONOMY
),
android_article_time_raw AS (
    SELECT
        DATE_TRUNC('day',  s.ORIGINAL_TIMESTAMP)::DATE          AS day,
        DATE_TRUNC('month', s.ORIGINAL_TIMESTAMP)::DATE         AS month,
        'app'                                                    AS platform,
        s.ARTICLE_ID::VARCHAR                                    AS story_id,
        COALESCE(NULLIF(TRIM(s.USER_ID),''), s.CONTEXT_DEVICE_ID) AS visitor_key,
        CASE WHEN NULLIF(TRIM(s.USER_ID),'') IS NOT NULL
             THEN NULLIF(TRIM(s.USER_ID),'')
             ELSE CONCAT('android:', s.CONTEXT_DEVICE_ID) END   AS monthly_key,
        s.ORIGINAL_TIMESTAMP,
        s.PARENT_TAXONOMY,                                       -- ← ADD THIS
        DATEDIFF('second',
            s.ORIGINAL_TIMESTAMP,
            MIN(t.ORIGINAL_TIMESTAMP)
        )                                                        AS time_on_screen_seconds
    FROM SEGMENT_EVENTS.RACING_POST_JANUS_ANDROID_LIVE.SCREEN_ARTICLE s
    JOIN SEGMENT_EVENTS.RACING_POST_JANUS_ANDROID_LIVE.TRACKS t
        ON  COALESCE(NULLIF(TRIM(t.USER_ID),''), t.CONTEXT_DEVICE_ID)
          = COALESCE(NULLIF(TRIM(s.USER_ID),''), s.CONTEXT_DEVICE_ID)
        AND t.ORIGINAL_TIMESTAMP > s.ORIGINAL_TIMESTAMP
        AND DATEDIFF('minute', s.ORIGINAL_TIMESTAMP, t.ORIGINAL_TIMESTAMP) <= 60
    WHERE s.ORIGINAL_TIMESTAMP::DATE BETWEEN (SELECT start_date FROM params) AND (SELECT end_date FROM params)
    AND s.ARTICLE_ID IS NOT NULL
    AND s.PARENT_TAXONOMY IN (SELECT v FROM app_parent_tax_allowlist)
    GROUP BY 1,2,3,4,5,6,s.ORIGINAL_TIMESTAMP,s.PARENT_TAXONOMY  -- ← ADD s.PARENT_TAXONOMY

    m.editorial_subscribers,
    m.subscriber_flag,
    IFF(m.grain='month' AND m.platform IN ('web','app') AND m.desk IN ('ALL_EDITORIAL','All Editorial'),
        um.daily_avg_pct_ultimate_consuming_editorial,NULL)::NUMBER(18,2) AS daily_avg_pct_ultimate_consuming_editorial,
    IFF(m.grain='month' AND m.platform IN ('web','app') AND m.desk IN ('ALL_EDITORIAL','All Editorial'),
        tm.daily_avg_pct_tipping_consuming_editorial,NULL)::NUMBER(18,2) AS daily_avg_pct_tipping_consuming_editorial,
    IFF(m.grain='day',
        ROUND(AVG(m.pageviews) OVER (
            PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
            ORDER BY m.period_start ROWS BETWEEN 12 PRECEDING AND CURRENT ROW),2),
        m.pageviews)::NUMBER(18,0)                                      AS pageviews,
    IFF(m.grain='month' AND m.desk IN ('ALL_EDITORIAL','All Editorial')
        AND COALESCE(m.editorial_subscribers,0)>0,
        ROUND(ad.total_active_days/NULLIF(m.editorial_subscribers,0),2),
        NULL)::NUMBER(18,2)                                             AS active_days_per_editorial_subscriber,
    m.unique_sessions,
    ROUND(m.pageviews::FLOAT/NULLIF(m.unique_sessions,0),2)::NUMBER(18,2) AS pageviews_per_unique_session,
    IFF(m.grain='day',
        ROUND(AVG(m.pageviews::FLOAT/NULLIF(m.unique_sessions,0)) OVER (
            PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
            ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),2),
        NULL)::NUMBER(18,2)                                             AS pageviews_per_session_dow_avg,
    IFF(m.grain='day',
        ROUND((m.pageviews::FLOAT/NULLIF(m.unique_sessions,0))
            -AVG(m.pageviews::FLOAT/NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),2),
        NULL)::NUMBER(18,2)                                             AS pageviews_vs_dow_avg,
    IFF(m.grain='day',
        ROUND(((m.pageviews::FLOAT/NULLIF(m.unique_sessions,0))
            -AVG(m.pageviews::FLOAT/NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING))
            /NULLIF(AVG(m.pageviews::FLOAT/NULLIF(m.unique_sessions,0)) OVER (
                PARTITION BY m.grain,m.platform,m.persona,m.desk,DAYOFWEEK(m.period_start)
                ORDER BY m.period_start ROWS BETWEEN 13 PRECEDING AND 1 PRECEDING),0)*100,2),
        NULL)::NUMBER(18,2)                                             AS pageviews_pct_vs_dow_avg,
    m.premium_editorial_users,
    m.free_editorial_users,
    -- ↓ FIXED ORDER: these 4 now match the view header column order ↓
    TO_VARCHAR(cs.editorial_clicks_by_source)                           AS editorial_clicks_by_source,
    TO_VARCHAR(fs.ftds_by_source)                                       AS ftds_by_source,
    COALESCE(
        TRY_TO_NUMBER(
            GET(TRY_PARSE_JSON(TO_VARCHAR(cs.editorial_clicks_by_source)),'TC_ARTICLE_CLICK')::STRING
        ),0
    )                                                                   AS tc_clicks,
    ftd.source                                                          AS ftd_source,
    tr.tc_traffic_events,                                               
    tr.rc_traffic_events,
    apvs.avg_pageviews_per_story                                        AS avg_pageviews_per_story,
    aets.avg_engaged_time_per_story                             AS avg_engaged_time_per_story  

FROM (
    SELECT * FROM total_metrics
    UNION ALL SELECT * FROM editorial_metrics_by_desk
    UNION ALL SELECT * FROM editorial_metrics_all_editorial
) m
LEFT JOIN net_subscriber_calc n
    ON n.grain=m.grain AND n.period_start=m.period_start AND n.platform=m.platform AND n.persona=m.persona AND n.desk=m.desk
LEFT JOIN active_days_all ad
    ON  ad.grain        = m.grain
    AND ad.period_start = m.period_start
    AND ad.platform     = m.platform
    AND ad.persona      = m.persona
    AND ad.desk         = m.desk
LEFT JOIN ftd_all ftd
    ON ftd.grain=m.grain AND ftd.period_start=m.period_start AND ftd.platform=m.platform AND ftd.persona=m.persona
    AND ftd.desk=CASE
        WHEN m.desk IN ('ALL_EDITORIAL','All Editorial') THEN 'All Editorial'
        WHEN m.desk IN ('news','News')                   THEN 'news'
        WHEN m.desk IN ('rp_recommends','RP Recommends') THEN 'rp_recommends'
        ELSE m.desk END
LEFT JOIN clicks_all cl
    ON cl.grain=m.grain AND cl.period_start=m.period_start AND cl.platform=m.platform AND cl.persona=m.persona AND cl.desk=m.desk
LEFT JOIN traffic_all tr
    ON tr.grain=m.grain AND tr.period_start=m.period_start AND tr.platform=m.platform AND tr.persona=m.persona AND tr.desk=m.desk
LEFT JOIN ultimate_month_daily_avg um
    ON um.period_start=m.period_start AND um.platform=m.platform
LEFT JOIN tipping_month_daily_avg tm
    ON tm.period_start=m.period_start AND tm.platform=m.platform
LEFT JOIN traffic_visitors_all tv
    ON tv.grain=m.grain AND tv.period_start=m.period_start AND tv.platform=m.platform AND tv.persona=m.persona AND tv.desk=m.desk
LEFT JOIN clicks_source_json cs
    ON cs.grain=m.grain AND cs.period_start=m.period_start AND cs.platform=m.platform AND cs.persona=m.persona AND cs.desk=m.desk
LEFT JOIN ftd_source_json fs
    ON fs.grain=m.grain AND fs.period_start=m.period_start AND fs.platform=m.platform AND fs.persona=m.persona
    AND fs.desk=CASE WHEN m.desk IN ('ALL_EDITORIAL','All Editorial') THEN 'All Editorial' ELSE m.desk END
LEFT JOIN avg_pv_per_story_all apvs
    ON  apvs.grain        = m.grain
    AND apvs.period_start = m.period_start
    AND apvs.platform     = m.platform
    AND apvs.persona      = m.persona
    AND apvs.desk         = m.desk
LEFT JOIN avg_engaged_time_all aets
    ON  aets.grain        = m.grain
    AND aets.period_start = m.period_start
    AND aets.platform     = m.platform
    AND aets.persona      = m.persona
    AND aets.desk         = m.desk
ORDER BY grain, period_start, platform, persona, desk;
